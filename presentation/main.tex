\section{Background}


\begin{frame}{GEMM Breakdown}
  A GEMM kernel has two major phases: \textbf{compute} and \textbf{epilogue}.

  \vspace{0.35cm}
  \begin{columns}[T,onlytextwidth]
    \column{0.48\textwidth}
    \textbf{Compute}
    \begin{itemize}
      \item Fetch A/B tiles.
      \item Stage tiles in shared memory.
      \item Issue MMA instructions.
      \item Accumulate results in TMEM.
    \end{itemize}

    \column{0.48\textwidth}
    \textbf{Epilogue}
    \begin{itemize}
      \item Load TMEM into registers.
      \item Apply optional epilogue fusion operations.
      \item Pack registers into shared memory.
      \item Store the final result to HBM.
    \end{itemize}
  \end{columns}
\end{frame}

\section{Orchestration}

\begin{frame}{The Core Idea: Orchestration}
  If there is one idea to take away from this talk, it is this:

  \vspace{0.35cm}
  \begin{center}
    \textbf{A high-performance GEMM kernel is an orchestration problem.}
  \end{center}

  \vspace{0.35cm}
  The kernel is a dataflow schedule over hardware units, warp roles, and
  on-chip containers. 
\end{frame}

\begin{frame}{Operations and Containers}
  We model the kernel as \textbf{operations} connected by \textbf{containers}
  that hold values:

  \vspace{0.5cm}

\begin{frame}{When Can an Operation Start?}
  An operation is schedulable only when both conditions are true:

  \vspace{0.25cm}
  \begin{enumerate}
    \item \textbf{Input value-ready}: every input value it needs has been produced.
    \item \textbf{Output resource-free}: the container for its output is available.
  \end{enumerate}

  \vspace{0.3cm}
  In other words, the operation is blocked if either the data is not ready
  or the destination container is still occupied.

  \vspace{0.25cm}
  This is the basic rule behind SMEM ring-buffer reuse, TMEM reuse, and TMA
  store-buffer pipelining.
\end{frame}

\begin{frame}{Two Signals}
  In this model, synchronization has two meanings:

  \begin{enumerate}
    \item \textbf{Data-ready}: a container is full; data is ready to be consumed.
    \item \textbf{Resource-free}: a container is empty; it can be reused.
  \end{enumerate}

  \vspace{0.3cm}
  A producer is waiting for a free output container. A consumer is waiting for
  a ready input value. Good GEMM scheduling is mostly about making those two
  waits overlap with useful work.

  \vspace{0.15cm}
  For TMEM, value readiness is gated by a completion condition: the
  \textbf{TMEM data ready} signal is sent only after the final MMA finishes.

  \vspace{0.25cm}

  for (int k = 0; k < num_k; k++) {

}

\section{2-CTA MMA}

\begin{frame}{One Key Optimization: 2-CTA MMA}
  \begin{columns}[T,onlytextwidth]

\begin{frame}{Epilogue Design: Multiple Store Buffers}
  With $\ge 2$ store buffers, one chunk's \textbf{TMA store overlaps the next
  chunk's SMEM write}:

  \vspace{0.3cm}

        wr/.style={draw, fill=orange!25, minimum height=0.5cm, font=\tiny, anchor=west, inner sep=1pt, align=center},

      \node[lane] at (-0.1,-0.55) {buffer 1};

      \node[st, minimum width=2.9cm] at (1.6,0.55) {TMA store $c_0 \to$ HBM};
      \node[wr, minimum width=1.5cm] at (4.5,0.55) {write $c_2$};
      \node[wr, minimum width=1.5cm] at (1.6,-0.55) {write $c_1$};
      \node[st, minimum width=2.9cm] at (3.1,-0.55) {TMA store $c_1 \to$ HBM};
      \draw[-{Latex[length=2mm]}, thick] (0,-1.15) -- (6.4,-1.15) node[right, font=\scriptsize]{time};
      \draw[densely dashed, draw=mmsignal] (1.6,0.95) -- (1.6,-1.0);
      \draw[densely dashed, draw=mmsignal] (3.1,0.95) -- (3.1,-1.0);
      \node[font=\scriptsize, text=mmsignal, align=center] at (2.35,1.35)
        {write $c_1$ runs while\\ TMA drains $c_0$};
    \end{tikzpicture}
  \end{center}

  \vspace{0.2cm}
  \begin{itemize}
    \item The TMA store is \textbf{async}: once issued, the epilogue moves to the
      next chunk's write.
    \item With $N$ buffers ($N \in \{1,2,3,4\}$) it fills a free one while TMA
      drains earlier ones --- stalling only when all $N$ are in flight.
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Epilogue Design: From Design to Code}
  The whole epilogue is one loop over the $BN/64$ store chunks:
  \begin{lstlisting}[language=C, basicstyle=\ttfamily\scriptsize]
for (int chunk = 0; chunk < BN/64; chunk++) {
  // Phase 1: load all 64 cols of this chunk into registers
  for (int n = 0; n < 8; n++)            // 8 x tcgen05.ld.x8
    tcgen05_ld_x8(tmem_row + chunk*64 + n*8, reg[n]);
  tcgen05_wait_ld();
  if (elect) tma_wait_group<TMA_STORE_STAGES-1>();  // a buffer is free
  bar.sync;                              // ... visible to all warps
  // Phase 2: conflict-free SMEM write, then the TMA store
  for (int n = 0; n < 8; n++) {
    int swz = n ^ (row & 7);             // XOR swizzle: distinct banks
    store_buf[stage][row][swz] = pack_bf16(reg[n]);
  }
  bar.sync;                              // buffer fully written
  if (elect) tma_store(store_buf[stage], C + chunk*64);
  stage = (stage + 1) % NUM_STORE_BUFFERS;
}
  \end{lstlisting}
\end{frame}

\section{Tile Sizes and Shared Memory}

\begin{frame}{Partitioning Shared Memory}
  \begin{center}
    \begin{tikzpicture}[
        seg/.style={draw, thick, minimum height=1.0cm, align=center, font=\small}
      ]
      \node[seg, fill=teal!12, minimum width=6.0cm] (comp) at (0,0)
        {compute ring\\ \scriptsize $NS$ slots $\times$ (A + B) tiles};
      \node[seg, fill=orange!14, minimum width=3.0cm, right=2pt of comp] (epi)
        {epilogue ring\\ \scriptsize store slots $\times$ buffer};
    \end{tikzpicture}
  \end{center}

  \vspace{0.1cm}
  \begin{itemize}
    \item \textbf{Compute ring}: $NS$ pipeline slots, each holding one A tile and
      one B tile for the MMA stage.
    \item \textbf{Epilogue ring}: store slots, each a TMA store buffer holding one
      $BM\times64$ bf16 chunk: $128\cdot64\cdot2 = 16$\,KB.
    \item Both rings share one per-SM SMEM budget, so larger tiles or deeper
      pipelines on one side leave less room for the other.
  \end{itemize}
\end{frame}

\begin{frame}{Tile Sizes and the SMEM Budget}
  Per compute slot (2-CTA, each CTA owns half of B):
  \[
    \text{slot} = \underbrace{BM\cdot BK\cdot 2}_{\text{A tile}}
                + \underbrace{\tfrac{BN}{2}\cdot BK\cdot 2}_{\text{B tile (local)}}
  \]

  \begin{table}
    \centering\small
    \begin{tabular}{lrrr}
      \toprule
      Config (2-CTA, $BK{=}64$) & A tile & B tile & slot \\
      \midrule
      $BM{=}128,\ BN{=}256$ & 16\,KB & 16\,KB & 32\,KB \\
      $BM{=}128,\ BN{=}512$ & 16\,KB & 32\,KB & 48\,KB \\
      \bottomrule
    \end{tabular}
  \end{table}

  \vspace{0.15cm}
  Budget constraint (per SM, B200 $\approx$ 227\,KB):
  \[
    \underbrace{NS \times \text{slot}}_{\text{compute ring}}
    \;+\;
    \underbrace{E \times 16\,\text{KB}}_{\text{epilogue ring}}
    \;\le\; S_{\text{SMEM}}
  \]

  \footnotesize Example: $BN{=}256$, $NS{=}5 \Rightarrow 160$\,KB compute ring;
  with $E{=}2$ store buffers, $+32$\,KB $= 192$\,KB total.
\end{frame}

  Each split below is the \textbf{maximum} that fits in $\approx$227\,KB; the

\section{Implementation \& Performance}

\begin{frame}[plain]
  \begin{center}
    \vspace{1.8cm}
    {\usebeamercolor[fg]{title}\Large\bfseries Implementation and Performance Results}
  \end{center}
\end{frame}

\begin{frame}{Implementation Details Worth Noting}
  \textbf{Warp layout --- deliberately not a power of two.} Each CTA launches
  \[
    \underbrace{1}_{\text{TMA}} + \underbrace{1}_{\text{MMA}}
    + \underbrace{4 \text{ or } 8}_{\text{epilogue}}
    \;=\; 6 \text{ or } 10 \text{ warps.}
  \]
  \begin{itemize}
    \item The \textbf{TMA} and \textbf{MMA} warps are fixed --- one each.
    \item The \textbf{epilogue} warp count $\in \{4, 8\}$ is \textbf{autotuned};
      more warps give the TMEM drain and SMEM write more parallelism.
    \item So the CTA is sized to the work, not rounded up to a power of two.
  \end{itemize}
\end{frame}

\begin{frame}{Implementation: Folding Store into the Epilogue}
  The design has four stages, but the implementation \textbf{merges the last
  two}: the epilogue warps also issue the TMA store --- no dedicated store warp,
  no producer/consumer handshake.

  \begin{itemize}
    \item \textbf{Leaner code, simpler synchronization} --- at the cost of a
      slightly less ``clean'' design.
    \item \textbf{No performance loss}: issuing a store is a few instructions (the
      transfer is async), and the epilogue is store-bound, so a dedicated issue
      warp would not raise TMA throughput.
  \end{itemize}

  \vspace{0.25cm}
  \emph{Why not also merge TMA-load with MMA, then?} Warp specialization pays off
  only when two roles have heavy, \emph{independent} async work to overlap. TMA
  load and MMA do --- overlapping them keeps the tensor cores fed. The store
  \emph{issue} is a handful of instructions gated by the same SMEM buffer and HBM
  bandwidth, so giving it its own warp buys nothing.
\end{frame}

\begin{frame}{Why More Buffers Doesn't Always Help}
  The autotuned configs rarely max out the buffer counts (store mostly 1--2, not
  4) --- something we would not have predicted before seeing the numbers. Some
  intuition (\emph{hypotheses}, not proven):

  \vspace{0.25cm}
  \textbf{Compute ring (more $NS$)}
  \begin{itemize}
    \item Once the pipeline hides load/MMA latency, extra stages add nothing.
  \end{itemize}

  \vspace{0.15cm}
  \textbf{Epilogue ring (more store buffers)}
  \begin{itemize}
    \item If the epilogue is already fast relative to compute, the extra
      store$\leftrightarrow$write overlap buys little.
    \item More TMA stores in flight $=$ more memory parallelism that may contend
      with the TMA \emph{loads} feeding the compute pipeline.
  \end{itemize}

  \vspace{0.2cm}
  \centering\small The optimum is shape-dependent and empirical --- hence autotune.
\end{frame}

\section{Adding Epilogue Fusions}

\begin{frame}[plain]
  \begin{center}
    \vspace{1.8cm}
    {\usebeamercolor[fg]{title}\Large\bfseries Adding Epilogue Fusions}
  \end{center}
\end{frame}

\begin{frame}{Adding Epilogue Fusions}
  A custom epilogue fusion is almost entirely an \textbf{epilogue} change --- the
  compute pipeline is reused, with at most a minor TMA-load tweak.

  \vspace{0.3cm}
  \textbf{Case study: fused SwiGLU.} One GEMM produces two projections side by
  side in TMEM (\emph{left} and \emph{gate}); the epilogue writes out both and
  the gated result:
  \[
    \text{SwiGLU} = \text{left} \;\odot\; \mathrm{SiLU}(\text{gate}),
    \qquad \mathrm{SiLU}(g) = g\,\sigma(g).
  \]
  \begin{itemize}
    \item Loop over \emph{pairs} of column chunks (a left chunk $+$ its gate chunk).
    \item Per pair: load both into registers, then \textbf{three} stores ---
      left, gate, and SwiGLU.
    \item \emph{Dual-B}: left and gate are \emph{separate} weight matrices; with
      2-CTA on, each CTA loads its half from a different one --- a tiny load
      change, with \textbf{no measured performance impact}.
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Fused SwiGLU: the Epilogue Loop}
  Only the epilogue changes; \code{stage\_and\_store} is the canonical
  swizzled-write $+$ TMA-store helper:
  \begin{lstlisting}[language=C, basicstyle=\ttfamily\scriptsize]
for (int p = 0; p < (BN/64)/2; p++) {          // loop over chunk PAIRS
  for (int n = 0; n < 8; n++) {                // load both projections
    tcgen05_ld_x8(tmem + p*64        + n*8, left[n]);
    tcgen05_ld_x8(tmem + (p+HALF)*64 + n*8, gate[n]);
  }
  tcgen05_wait_ld();

  stage_and_store(left, C, p*64);              // (1) store left
  stage_and_store(gate, C, (p+HALF)*64);       // (2) store gate

  for (int n = 0; n < 8; n++)                  // (3) fused SwiGLU math
    swig[n] = left[n] * silu_fast(gate[n]);    //     ex2.approx + rcp.approx
  stage_and_store(swig, D, p*64);              //     store SwiGLU -> D
}
  \end{lstlisting}
  \vspace{-0.15cm}
  {\footnotesize \code{stage\_and\_store} = the same store step as the canonical
  epilogue; fusion just adds the math and an extra output tensor \code{D}.}
\end{frame}

\begin{frame}{Fused SwiGLU: Almost Free}
  \begin{center}\small
    $M{=}32768,\ N{=}4608,\ K{=}768$; \quad $BM{=}128,\ BN{=}256,\ NS{=}6,\
    \text{store}{=}2$
  \end{center}

  \vspace{0.15cm}
  \begin{center}\small
    \begin{tabular}{lrr}
      \toprule
       & time & TFLOPS \\
      \midrule
      cuBLAS GEMM (matmul only)              & 165\,$\mu$s & 1406 \\
      unfused: GEMM $+$ separate SwiGLU      & $\sim$245\,$\mu$s & --- \\
      \quad\scriptsize ($165 + {\sim}80$\,$\mu$s activation) & & \\
      \textbf{fused SwiGLU} (one kernel)     & \textbf{185}\,$\mu$s & 1250 \\
      \bottomrule
    \end{tabular}
  \end{center}

  \vspace{0.2cm}
  \begin{itemize}
    \item vs the \textbf{unfused two-kernel} path ($\sim$245\,$\mu$s), fusion is
      \textbf{$\sim$1.3$\times$ faster} ($\sim$60\,$\mu$s saved).
    \item vs a \textbf{pure cuBLAS GEMM} (165\,$\mu$s), only $\sim$12\% slower
      despite $1.5\times$ store $+$ SwiGLU math --- the math is \textbf{free}, so
      the gap is just the extra store.
  \end{itemize}
\end{frame}

\begin{frame}{Future Direction: Epilogue Fusion as a Pattern}
  SwiGLU is just one instance. The epilogue template could expose a general
  \textbf{fusion hook}, run per $128\times64$ chunk:

  \vspace{0.15cm}
  \begin{enumerate}
    \item optionally \textbf{load extra inputs} into the chunk buffer (e.g. a
      residual or bias from HBM);
    \item apply \textbf{custom math} to the TMEM chunk (and those inputs);
    \item emit \textbf{one or more stores}.
  \end{enumerate}

  \vspace{0.25cm}
  The framework keeps supplying the hard parts --- TMEM drain, XOR swizzle,
  staged TMA store, pipelining --- and the user writes only the fusion. That is
  effectively a small \textbf{DSL for epilogue fusions}: bias$+$activation,
  residual add, gated activations, quantization, \ldots
\end{frame}
