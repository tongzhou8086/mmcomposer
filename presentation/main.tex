\section{GEMM and B200 Background}
\sectionslide{toc:bg}{GEMM and B200 Background}

\begin{frame}{GEMM: the Core Operation}
  A \textbf{GEMM} (general matrix multiply) computes
  \[
    \underbrace{C}_{M\times N} \;=\; \underbrace{A}_{M\times K}\,\cdot\,
    \underbrace{B}_{K\times N},
    \qquad C_{ij} = \sum_{k=1}^{K} A_{ik}\,B_{kj}.
  \]
  \begin{itemize}
    \item It is the workhorse of ML --- every linear layer and attention
      projection is a GEMM.
    \item Three sizes: \textbf{$M$} (output rows), \textbf{$N$} (output columns),
      \textbf{$K$} (the contracted ``inner'' dimension).
    \item Each of the $M\times N$ outputs sums $K$ products --- a lot of reuse of
      $A$ and $B$, which is what makes tiling pay off.
  \end{itemize}
\end{frame}

\begin{frame}{Tiling: Where $BM$, $BN$, $BK$ Come From}
  The output is computed in \textbf{tiles}. One CTA (thread block) owns one
  $BM\times BN$ output tile and sweeps $K$ in $BK$-wide chunks:

  \vspace{0.05cm}

      \node[dim] at (1.4,-0.35) {$C\;[M\times N]$};

      \draw[{Latex}-{Latex}, mmgray] (-2.1,2.32) -- (-1.675,2.32);

      \draw[{Latex}-{Latex}, mmgray] (0.7,2.32) -- (1.4,2.32);

  \vspace{0.05cm}
  \begin{itemize}
    \item \textbf{$BM\times BN$} --- the output tile one CTA produces (a row band
      of $A$ $\times$ a column band of $B$).
    \item \textbf{$BK$} --- the $K$-slice loaded per step (an $A$ tile
      $BM{\times}BK$ and a $B$ tile $BK{\times}BN$); loop over the $K/BK$ chunks.
  \end{itemize}
\end{frame}

\begin{frame}{The B200 Streaming Multiprocessor (SM)}
  \begin{columns}[c,onlytextwidth]
    \column{0.44\textwidth}
    \begin{center}
      \includegraphics[height=0.80\textheight]{images/b200_arch.png}
    \end{center}

    \column{0.56\textwidth}
    The pieces this talk relies on:
    \begin{itemize}
      \item \textbf{TMA} (Tensor Memory Accelerator) --- bulk
        HBM\,$\leftrightarrow$\,SMEM tile moves.
      \item \textbf{256\,KB L1 / shared memory} --- the compute and store rings.
      \item \textbf{5th-gen Tensor Cores} $+$ \textbf{256\,KB TMEM}
        (4 sub-partitions $\times$ 64\,KB each) --- the MMA and its accumulator.
      \item \textbf{CUDA cores} $+$ \textbf{register file} --- run the epilogue.
    \end{itemize}

    \vspace{0.2cm}
    {\footnotesize Four sub-partitions per SM, each with its own warp scheduler,
    registers, TMEM, and CUDA / tensor cores.}
  \end{columns}
\end{frame}

\begin{frame}{Threads, Warps, CTAs, Clusters}
  The GPU execution hierarchy this talk leans on:
  \begin{itemize}
    \item A \textbf{warp} --- 32 threads issued together; the unit of scheduling.
    \item A \textbf{CTA} (thread block) --- several warps sharing one SM's shared
      memory.
    \item A \textbf{cluster} --- a few CTAs on neighbouring SMs that can read each
      other's SMEM. We use \textbf{2-CTA} clusters.
  \end{itemize}

  \vspace{0.25cm}
  \textbf{Warp specialization.} Rather than every warp doing the same work, we
  give warps \emph{different roles} --- a load warp, an MMA warp, epilogue warps
  --- so the independent hardware units (TMA engine, tensor cores, CUDA cores)
  can all stay busy at once. That overlap is the whole game.
\end{frame}

  \vspace{0.25cm}
  \small One \textbf{descriptor} names the tensor ($A$: $M\times K$) + strides and
  the tile's corner $(x,y)$ and size $BM\times BK$ (its \code{boxDim}). The engine
  runs \textbf{async} (completion via an \code{mbarrier}), lands the tile
  \textbf{swizzled}, and can \textbf{multicast} to several CTAs.
  (\code{cp.async} is the older, per-thread (SIMT) alternative.)
  TMA boxes can be \textbf{1D--5D}; matmul uses the \textbf{2D} form shown here.
\end{frame}

\begin{frame}{mbarrier: a Shared-Memory Barrier}
  An \textbf{mbarrier} is a single \textbf{64-bit (8-byte)} word in shared memory.
  The hardware packs all of its synchronization state into that one word:

  \vspace{0.35cm}
  \begin{center}
  \begin{tikzpicture}[font=\scriptsize, scale=0.95, transform shape,
      f/.style={draw, thick, minimum height=1.0cm, align=center}]
    \node[f, fill=blue!12, minimum width=1.1cm] (ph) at (0,0) {phase\\(parity)};
    \node[f, fill=orange!16, minimum width=2.6cm, anchor=west] (ar) at (ph.east) {pending\\arrival count};
    \node[f, fill=green!16, minimum width=2.8cm, anchor=west] (tx) at (ar.east) {pending tx-count\\(bytes)};
    \node[font=\scriptsize, text=mmgray] at (3.25,0.85) {one 64-bit word in shared memory};
  \end{tikzpicture}
  \end{center}

  \vspace{0.5cm}
  \small \textbf{Common operations:} \code{init(N)} sets the arrival count;
  \code{arrive} checks a thread in (count\,$-1$); \code{arrive.expect\_tx(bytes)}
  checks in \emph{and} arms the byte count; \code{try\_wait.parity} blocks until
  the \textbf{phase flips} --- which the hardware does once both counts reach zero.

  \vspace{0.15cm}
  \small \textbf{Typical use:} signal that data is \emph{ready to consume}, or
  that a buffer is \emph{free to reuse}.
\end{frame}

\begin{frame}{One Framework, Not the Only Way}
  There are many valid ways to build a GEMM kernel on B200. This talk presents
  \textbf{one} --- a \textbf{pipelining-first}, warp-specialized design.

  \vspace{0.2cm}
  \begin{itemize}
    \item Many choices are illustrative, not mandatory. The epilogue here uses a
      \textbf{TMA store} --- chosen to show \emph{store pipelining} --- but a
      plain vectorized 16-byte global store works just as well.
    \item Likewise the input load uses \textbf{TMA}, but \code{cp.async} is also
      async and would work --- TMA load is just more B200-native.
  \end{itemize}

  \vspace{0.15cm}
  It is a \textbf{canonical default} that empirically performs well across most
  shapes --- not the only, nor always the optimal, design.
\end{frame}

\section{Pipelining: Overlap and Synchronization}
\sectionslide{toc:pipe}{Pipelining: Overlap and Synchronization}

\begin{frame}{The Core Idea: Orchestration}
  If there is one idea to take away from this talk, it is this:

  \vspace{0.3cm}
  \begin{center}
    \textbf{A high-performance GEMM kernel is a \emph{pipelining} problem.}
  \end{center}

  \vspace{0.3cm}
  \begin{itemize}
    \item \textbf{Overlap} the engines --- TMA, tensor cores, CUDA cores run
      \textbf{asynchronously}, so none sits idle.
    \item \textbf{Synchronize} only when needed, on specific \emph{events} --- a
      value is ready to be consumed, a container is free to be reused.
  \end{itemize}

  \vspace{0.2cm}
  We call this scheduling \textbf{orchestration}: a dataflow over hardware units,
  warp roles, and on-chip containers.
\end{frame}

\begin{frame}{Operations and Containers}
  We model the kernel as \textbf{operations} connected by \textbf{containers}
  that hold values:

  \vspace{0.5cm}
  \begin{center}
    \begin{tikzpicture}[
        box/.style={draw, thick, rounded corners=2pt, minimum width=2.7cm, minimum height=0.9cm, align=center},
        arrow/.style={-{Latex[length=2mm]}, thick}
      ]
      \node[box, fill=blue!8] (in) {Input\\containers};
      \node[box, fill=green!10, right=1.1cm of in] (op) {Operation};
      \node[box, fill=orange!12, right=1.1cm of op] (out) {Output\\container};
      \draw[arrow] (in) -- node[above]{drain} (op);
      \draw[arrow] (op) -- node[above]{fill} (out);
    \end{tikzpicture}
  \end{center}

  \vspace{0.5cm}
  \begin{itemize}
    \item An operation \textbf{drains} input values from its input containers.
    \item It \textbf{fills} the result into one output container.
  \end{itemize}
\end{frame}

\begin{frame}{When Can an Operation Start?}
  An operation is schedulable only when both conditions are true:

  \vspace{0.25cm}
  \begin{enumerate}
    \item \textbf{Input value-ready}: every input value it needs has been produced.
    \item \textbf{Output resource-free}: the container for its output is available.
  \end{enumerate}

  \vspace{0.3cm}
  In other words, the operation is blocked if either the data is not ready
  or the destination container is still occupied.

  \vspace{0.25cm}
  There are two kinds of synchronization signals per buffer: \textbf{data ready}
  and \textbf{buffer free}.
\end{frame}

\begin{frame}{One Key Optimization: 2-CTA MMA}
  \begin{columns}[T,onlytextwidth]
    \column{0.43\textwidth}
    \vspace{0.35cm}
    \begin{itemize}
      \item Two CTAs form one cluster.
      \item Each CTA keeps one A slice and half of B in SMEM.
      \item One group-2 MMA produces a larger, higher-AI output tile.

    \small

    larger tile while B storage is split across the CTAs.

\begin{frame}[fragile]{Epilogue Design: From Design to Code}
  The whole epilogue is one loop over the $BN/64$ store chunks:
  \begin{lstlisting}[language=C, basicstyle=\ttfamily\scriptsize]
for (int chunk = 0; chunk < BN/64; chunk++) {
  // Phase 1: load all 64 cols of this chunk into registers
  for (int n = 0; n < 8; n++)            // 8 x tcgen05.ld.x8
    tcgen05_ld_x8(tmem_row + chunk*64 + n*8, reg[n]);
  tcgen05_wait_ld();
  if (elect) tma_wait_group<TMA_STORE_STAGES-1>();  // a buffer is free
  bar.sync;                              // ... visible to all warps
  // Phase 2: conflict-free SMEM write, then the TMA store
  for (int n = 0; n < 8; n++) {
    int swz = n ^ (row & 7);             // XOR swizzle: distinct banks
    store_buf[stage][row][swz] = pack_bf16(reg[n]);
  }
  bar.sync;                              // buffer fully written
  if (elect) tma_store(store_buf[stage], C + chunk*64);
  stage = (stage + 1) % NUM_STORE_BUFFERS;
}
  \end{lstlisting}
\end{frame}

      \node[seg, fill=teal!12, minimum width=6.0cm] (comp) at (0,0)
        {compute ring\\ \scriptsize $NS$ slots $\times$ (A + B) tiles};
      \node[seg, fill=orange!14, minimum width=3.0cm, right=2pt of comp] (epi)
        {epilogue ring\\ \scriptsize store slots $\times$ buffer};
    \end{tikzpicture}
  \end{center}

  \vspace{0.1cm}
  \begin{itemize}
    \item \textbf{Compute ring}: $NS$ pipeline slots, each holding one A tile and
      one B tile for the MMA stage.
    \item \textbf{Epilogue ring}: store slots, each a TMA store buffer holding one
      $BM\times64$ bf16 chunk: $128\cdot64\cdot2 = 16$\,KB.
    \item Both rings share one per-SM SMEM budget, so larger tiles or deeper
      pipelines on one side leave less room for the other.
  \end{itemize}
\end{frame}

\begin{frame}{Tile Sizes and the SMEM Budget}
  Per compute slot (2-CTA, each CTA owns half of B):
  \[
    \text{slot} = \underbrace{BM\cdot BK\cdot 2}_{\text{A tile}}
                + \underbrace{\tfrac{BN}{2}\cdot BK\cdot 2}_{\text{B tile (local)}}
  \]

  \begin{table}
    \centering\small
    \begin{tabular}{lrrr}
      \toprule
      Config (2-CTA, $BK{=}64$) & A tile & B tile & slot \\
      \midrule
      $BM{=}128,\ BN{=}256$ & 16\,KB & 16\,KB & 32\,KB \\
      $BM{=}128,\ BN{=}512$ & 16\,KB & 32\,KB & 48\,KB \\
      \bottomrule
    \end{tabular}
  \end{table}

  \vspace{0.15cm}
  Budget constraint (per SM, B200 $\approx$ 227\,KB):
  \[
    \underbrace{NS \times \text{slot}}_{\text{compute ring}}
    \;+\;
    \underbrace{E \times 16\,\text{KB}}_{\text{epilogue ring}}
    \;\le\; S_{\text{SMEM}}
  \]

  \footnotesize Example: $BN{=}256$, $NS{=}5 \Rightarrow 160$\,KB compute ring;
  with $E{=}2$ store buffers, $+32$\,KB $= 192$\,KB total.
\end{frame}

\section{Implementation and Performance}
\sectionslide{toc:impl}{Implementation and Performance}

\begin{frame}{Implementation: Warp Layout}
  \textbf{Warp layout --- deliberately not a power of two.} Each CTA launches
  \[
    \underbrace{1}_{\text{TMA}} + \underbrace{1}_{\text{MMA}}
    + \underbrace{4 \text{ or } 8}_{\text{epilogue}}
    \;=\; 6 \text{ or } 10 \text{ warps.}
  \]
  \begin{itemize}
    \item The \textbf{TMA} and \textbf{MMA} warps are fixed --- one each.
    \item The \textbf{epilogue} warp count $\in \{4, 8\}$ is \textbf{autotuned};
      more warps give the TMEM drain and SMEM write more parallelism.
    \item So the CTA is sized to the work, not rounded up to a power of two.
  \end{itemize}
\end{frame}

\begin{frame}{Autotuning: Configuration Search Space}
  Autotune sweeps these parameters and benchmarks every \emph{valid} combo:

  \begin{table}
    \centering\small
    \begin{tabular}{ll}
      \toprule
      Parameter (covered in this talk) & Candidate values \\
      \midrule
      $BM$ (tile rows)               & $\{128\}$ \\
      $BN$ (tile cols)               & $\{256,\ 512\}$ \\
      compute slots ($NS$)           & $\{2,3,4,5,6\}$ \\
      epilogue slots (store buffers) & $\{1,2,3,4\}$ \\
      epilogue warps                 & $\{4,8\}$ \\
      \bottomrule
    \end{tabular}
  \end{table}

  \vspace{0.1cm}
  \begin{itemize}
    \item A \textbf{static checker} prunes combos whose SMEM exceeds
      \textbf{227\,KB} ($NS \times \text{slot}(BN) + \text{store}\times16
      \le 227$) before benchmarking; the fastest survivor wins.
    \item The framework fixes everything else; the one knob it leaves open is the
      \textbf{CTA-swizzle factor} $\{1,2,4,8,16\}$ (an L2-reuse block ordering,
      not covered here).
  \end{itemize}
\end{frame}

  \textbf{MMComposer} is a CUDA GEMM kernel \emph{generator}: pick a point in a
  space of design knobs and it splices hand-tuned fragments into a complete,

\begin{frame}{Why More Buffers Doesn't Always Help}
  The autotuned configs rarely max out the buffer counts (store mostly 1--2, not
  4) --- something we would not have predicted before seeing the numbers. Some
  intuition (\emph{hypotheses}, not proven):

  \vspace{0.25cm}
  \textbf{Compute ring (more $NS$)}
  \begin{itemize}
    \item Once the pipeline hides load/MMA latency, extra stages add nothing.
  \end{itemize}

  \vspace{0.15cm}
  \textbf{Epilogue ring (more store buffers)}
  \begin{itemize}
    \item If the epilogue is already fast relative to compute, the extra
      store$\leftrightarrow$write overlap buys little.
    \item More TMA stores in flight $=$ more memory parallelism that may contend
      with the TMA \emph{loads} feeding the compute pipeline.
  \end{itemize}

  \vspace{0.2cm}
  \centering\small The optimum is shape-dependent and empirical --- hence autotune.
\end{frame}

\section{Backup Slides}
\sectionslide{toc:backup}{Backup Slides}

\begin{frame}{Implementation: TMA Loading}
  Every K-step, a \textbf{single elected lane} in the TMA warp issues one
  bulk-tensor copy per operand (\code{cp.async.bulk.tensor.2d}, the
  \code{cta\_group::2} cluster form):
  \begin{itemize}
    \item \textbf{A tile} --- $BM\times BK = 128\times64$ bf16 $=16$\,KB, one
      descriptor copy.
    \item \textbf{B tile} --- $BN_{\text{local}}\times BK$, issued in $64$-column
      chunks; each CTA loads only its own $BN/2$ columns.
    \item SMEM destinations use the \textbf{128-byte swizzle}
      (\code{SWIZZLE\_128B}) --- one atom is $128$\,B $= 64$ bf16, which fixes the
      inner load box to $64$ --- so the subsequent MMA reads SMEM
      bank-conflict-free.
    \item \textbf{Data-ready signal.} The producer arms the slot's \code{mbarrier}
      with the expected byte count (\code{arrive.expect\_tx}, software) and
      \emph{continues without blocking}. The TMA \emph{engine} credits bytes as
      the data lands; once the count is met the hardware \textbf{auto-flips the
      barrier's phase}
  \end{itemize}
\end{frame}

\begin{frame}{Implementation: MMA Issue}
  The MMA warp drives the tensor core with one instruction
  (\code{tcgen05.mma.cta\_group::2.kind::f16}):
  \begin{itemize}
    \item \textbf{Shape} --- $N$ width is $64$, $128$, or $256$ (the max); $K=16$
      for bf16. So a $128\times256$ tile over $BK=64$ is \textbf{4 K-chained
      MMAs} ($64/16$).
    \item \textbf{One thread, one warp} --- a single elected lane issues each MMA;
      operands are named by \textbf{SMEM matrix descriptors}, not staged through
      registers.
    \item \textbf{2-CTA} --- only \textbf{rank 0} issues, but the tensor cores on
      \textbf{both} ranks compute (the instruction spans $M = 2\times128$ across
      the cluster).
    \item \textbf{Async completion signal.} Partial sums accumulate in TMEM.
      The lane issues the MMAs then \code{tcgen05.commit} and
      \emph{continues without blocking}; once all the MMAs finish, the hardware
      \textbf{auto-flips the barrier's phase} --- signalling ``TMEM data ready.''
  \end{itemize}
\end{frame}

\begin{frame}{Implementation: Signals via mbarrier}
  Each ready/free signal is one \code{mbarrier}. A producer issues or arms it and
  moves on; the \textbf{hardware auto-flips the phase} once the relevant work
  completes. The consumer blocks in \code{mbarrier\_wait\_phase}, which spins on
  \code{try\_wait.parity} until that flip:

  \vspace{0.15cm}
  \begin{columns}[T,onlytextwidth]
    \column{0.5\textwidth}
    \textbf{Data-ready (``full'')}
    \begin{itemize}
      \item TMA\,$\to$\,SMEM: \code{mbarrier\_arrive\_expect\_tx(.., SLOT\_BYTES)}
        --- flips once the bytes land.
      \item MMA\,$\to$\,TMEM: \code{tcgen05\_commit\_mcast\_g2(tmem\_full)} ---
        flips once prior MMAs finish.
    \end{itemize}

    \column{0.5\textwidth}
    \textbf{Resource-free}
    \begin{itemize}
      \item MMA\,$\to$\,SMEM slot: \code{tcgen05\_commit\_mcast\_g2(free\_mbar)}
        --- flips after prior MMAs finish.
      \item Epilogue\,$\to$\,TMEM: \code{EPI\_TMEM\_FREE\_ARRIVE} --- flips once
        the epilogue has drained TMEM.
    \end{itemize}
  \end{columns}

  \vspace{0.1cm}
  \footnotesize These are exactly the \code{mbarrier} calls in the warp-code
  slides --- the concrete form of the dashed ready/free arrows.
\end{frame}

\section{Adding a Custom Epilogue Fusion}
\sectionslide{toc:fusion}{Adding a Custom Epilogue Fusion}

\begin{frame}{Adding Epilogue Fusions}
  A custom epilogue fusion is almost entirely an \textbf{epilogue} change --- the
  compute pipeline is reused, with at most a minor TMA-load tweak.

  \vspace{0.3cm}
  \textbf{Case study: fused SwiGLU.} One GEMM produces two projections side by
  side in TMEM (\emph{left} and \emph{gate}); the epilogue writes out both and
  the gated result:
  \[
    \text{SwiGLU} = \text{left} \;\odot\; \mathrm{SiLU}(\text{gate}),
    \qquad \mathrm{SiLU}(g) = g\,\sigma(g).
  \]
  \begin{itemize}
    \item Loop over \emph{pairs} of column chunks (a left chunk $+$ its gate chunk).
    \item Per pair: load both into registers, then \textbf{three} stores ---
      left, gate, and SwiGLU.
    \item \emph{Dual-B}: left and gate are \emph{separate} weight matrices; with
      2-CTA on, each CTA loads its half from a different one --- a tiny load
      change, with \textbf{no measured performance impact}.
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Fused SwiGLU: the Epilogue Loop}
  Only the epilogue changes; \code{stage\_and\_store} is the canonical
  swizzled-write $+$ TMA-store helper:
  \begin{lstlisting}[language=C, basicstyle=\ttfamily\scriptsize]
for (int p = 0; p < (BN/64)/2; p++) {          // loop over chunk PAIRS
  for (int n = 0; n < 8; n++) {                // load both projections
    tcgen05_ld_x8(tmem + p*64        + n*8, left[n]);
    tcgen05_ld_x8(tmem + (p+HALF)*64 + n*8, gate[n]);
  }
  tcgen05_wait_ld();

  stage_and_store(left, C, p*64);              // (1) store left
  stage_and_store(gate, C, (p+HALF)*64);       // (2) store gate

  for (int n = 0; n < 8; n++)                  // (3) fused SwiGLU math
    swig[n] = left[n] * silu_fast(gate[n]);    //     ex2.approx + rcp.approx
  stage_and_store(swig, D, p*64);              //     store SwiGLU -> D
}
  \end{lstlisting}
  \vspace{-0.15cm}
  {\footnotesize \code{stage\_and\_store} = the same store step as the canonical
  epilogue; fusion just adds the math and an extra output tensor \code{D}.}
\end{frame}

\begin{frame}{Fused SwiGLU: Almost Free}
  \begin{center}\small
    $M{=}32768,\ N{=}4608,\ K{=}768$; \quad $BM{=}128,\ BN{=}256,\ NS{=}6,\
    \text{store}{=}2$
  \end{center}

  \vspace{0.15cm}
  \begin{center}\small
    \begin{tabular}{lrr}
      \toprule
       & time & TFLOPS \\
      \midrule
      cuBLAS GEMM (matmul only)              & 165\,$\mu$s & 1406 \\
      unfused: GEMM $+$ separate SwiGLU      & $\sim$245\,$\mu$s & --- \\
      \quad\scriptsize ($165 + {\sim}80$\,$\mu$s activation) & & \\
      \textbf{fused SwiGLU} (one kernel)     & \textbf{185}\,$\mu$s & 1250 \\
      \bottomrule
    \end{tabular}
  \end{center}

  \vspace{0.2cm}
  \begin{itemize}
    \item vs the \textbf{unfused two-kernel} path ($\sim$245\,$\mu$s), fusion is
      \textbf{$\sim$1.3$\times$ faster} ($\sim$60\,$\mu$s saved).
    \item vs a \textbf{pure cuBLAS GEMM} (165\,$\mu$s), only $\sim$12\% slower
      despite $1.5\times$ store $+$ SwiGLU math --- the math is \textbf{free}, so
      the gap is just the extra store.
  \end{itemize}
\end{frame}

\begin{frame}{Future Direction: Epilogue Fusion as a Pattern}
  SwiGLU is just one instance. The epilogue template could expose a general
  \textbf{fusion hook}, run per $128\times64$ chunk:

  \vspace{0.15cm}
  \begin{enumerate}
    \item optionally \textbf{load extra inputs} into the chunk buffer (e.g. a
      residual or bias from HBM);
    \item apply \textbf{custom math} to the TMEM chunk (and those inputs);
    \item emit \textbf{one or more stores}.
  \end{enumerate}

  \vspace{0.25cm}
  The framework keeps supplying the hard parts --- TMEM drain, XOR swizzle,
  staged TMA store, pipelining --- and the user writes only the fusion. That is
  effectively a small \textbf{DSL for epilogue fusions}: bias$+$activation,
  residual add, gated activations, quantization, \ldots
\end{frame}
